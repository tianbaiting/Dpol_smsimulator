\begin{frame}{A Polarized Deuteron as an Isovector Probe}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \small
      \begin{itemize}
        \item A deuteron is a bound proton--neutron pair.
        \item Polarization gives this pair a \emph{preferred orientation}
              relative to the beam.
        \item Near a neutron-rich target, proton and neutron feel
              \emph{different} isovector forces.
        \item After breakup, this appears as a bias in their
              \emph{relative} momentum.
      \end{itemize}
      \vspace{0.1cm}
      \begin{block}{Core idea}
        \footnotesize
        A polarized deuteron is an oriented proton--neutron probe of the
        isovector field.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.50\textheight,keepaspectratio]{d_pol_sche.png}\\[0.15cm]
      {\footnotesize\itshape Can this proton--neutron momentum bias be
      reconstructed at SAMURAI?}
    \end{column}
  \end{columns}
\end{frame}

% Slide 3 --- How the IVR Signal Is Formed
%=========================================================================
\begin{frame}{How the IVR Signal Is Formed}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\linewidth,keepaspectratio]{isovector_reorientaton.png}
    \end{column}
    \begin{column}{0.54\textwidth}
      \small
      \begin{itemize}
        \item Before interaction: an oriented, polarized deuteron.
        \item Near the target: proton and neutron feel different isovector
              forces.
        \item After breakup: a bias appears in their relative momentum
              $\Delta p_x$.
      \end{itemize}
        \[
    E_{\mathrm{sym}}(\rho)
    =
    E_{\mathrm{sym}}^{\mathrm{kin}}(\rho)
    + C\left(\frac{\rho}{\rho_0}\right)^{\gpar} .
  \]
      \begin{block}{Role of $\gpar$}
        \footnotesize
        In ImQMD, $\gpar$ sets the density dependence of the symmetry energy,
        and therefore the relative proton--neutron impulse; different $\gpar$
        predicts a different $\Delta p_x$ bias.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
%=========================================================================
% Slide 5 --- What Do We Measure?
%=========================================================================
\begin{frame}{What Do We Measure?}
  \small
  For each event we define a reaction plane from the proton and neutron
  directions, rotate into it, and compare who receives the larger momentum
  component along the selected axis:
  \[
    \Delta p_x = p_{x,p}-p_{x,n},
    \qquad\qquad
    R_x = \frac{N(\Delta p_x>0)}{N(\Delta p_x<0)} .
  \]
  \begin{block}{In words}
    \small
    We do not measure the force event by event.  We count whether the proton
    or the neutron receives the larger momentum component along the selected
    axis.  A larger $R_x$ means a stronger proton--neutron bias.
  \end{block}
  \vspace{0.1cm}
  \begin{itemize}
    \item $R_x>1$: the proton side is favoured.
    \item $R_x$ changes with the isovector force, and therefore with $\gpar$.
    \item In the detector-level analysis the proton momentum, neutron
          momentum, and event plane are all reconstructed.
  \end{itemize}
\end{frame}

% Slide 7 --- Why Start with y Polarization?
%=========================================================================
\begin{frame}{Why Start with y Polarization?}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.54\textwidth}
      \begin{block}{z-pol: cleaner physics, harder experiment}
        \scriptsize
        \begin{itemize}
          \item Polarization axis along the beam $\Rightarrow$ always in the
                reaction plane, no event-by-event dilution.
          \item But a longitudinal beam is hard to \emph{produce},
                \emph{transport} (spin precession), and \emph{monitor}
                ($p_{zz}$: absolute, two-angle normalization).
        \end{itemize}
      \end{block}
      \begin{block}{y-pol: measurable now}
        \scriptsize
        \begin{itemize}
          \item Tensor polarization is rank-2: the vertical axis projects onto
                the reaction plane with a $\cos(2\psi)$-like dependence.
          \item Event-by-event reaction-plane reconstruction lets us select a
                favorable $\psi$ window; LRUD $p_{yy}$ monitoring is mature and
                statistics afford the selection.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.04\textwidth}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{block}{Current detector-level scope}
        \scriptsize
        The reco-cut closure shown in this talk is the y-pol chain:
        reconstructed proton, neutron, event plane, fiducial, and quality cuts.
      \end{block}
      \begin{block}{z-pol status}
        \scriptsize
        z-pol remains a valuable extension, but its detector-level $R_x$ curve
        is not used here until the same reco-cut closure is completed.
      \end{block}
    \end{column}
  \end{columns}
  \vspace{-0.08cm}
  \begin{block}{Takeaway}
    \scriptsize\centering
    y-pol is the realistic first stage; z-pol is a valuable later extension.
  \end{block}
\end{frame}

        systematics.

%---- Backup A: IVR model ----
\begin{frame}{Backup A: Symmetry Energy and $\gpar$}
  \small
  The symmetry term in the ImQMD energy density is parametrized as
  \[
    E_{\mathrm{sym}}(\rho)
    =
    E_{\mathrm{sym}}^{\mathrm{kin}}(\rho)
    + C\left(\frac{\rho}{\rho_0}\right)^{\gpar} .
  \]
  \begin{itemize}
    \item $\gpar$ controls the density dependence around and below saturation.
    \item Larger $\gpar$ $\Rightarrow$ stiffer neutron--proton splitting at
          sub-saturation density.
    \item This changes the relative impulse on $p$ and $n$ near the target,
          and therefore the predicted $\Delta p_x$ bias.
  \end{itemize}
  \begin{block}{Grid used here}
    \small $\gpar = 0.5,\;0.6,\;0.7,\;0.8$ for both $^{112}$Sn and $^{124}$Sn.
  \end{block}
\end{frame}

%---- Backup B: observable definitions ----
\begin{frame}{Backup B: Observable Definitions}
  \small
  Reconstructed event plane:
  \[
    \phi_{\mathrm{event}}^{\,\mathrm{reco}}
    = \operatorname{atan2}\!\left(
      p_{y,p}^{\,\mathrm{reco}}+p_{y,n}^{\,\mathrm{reco}},\;
      p_{x,p}^{\,\mathrm{reco}}+p_{x,n}^{\,\mathrm{reco}}
    \right)
  \]
  Rotated relative momentum and ratio:
  \[
    \Delta p_{x,\mathrm{reco}}^{\,\mathrm{rot}}
    = p_{x,p,\mathrm{reco}}^{\,\mathrm{rot}}-p_{x,n,\mathrm{reco}}^{\,\mathrm{rot}},
    \qquad
    R_x^{\,\mathrm{reco,fid}}
    = \frac{N(\Delta p_{x,\mathrm{reco}}^{\,\mathrm{rot}}>0)}
           {N(\Delta p_{x,\mathrm{reco}}^{\,\mathrm{rot}}<0)} .
  \]
  Bounded asymmetry cross-check:
  \[
    A_x=\frac{N_+-N_-}{N_++N_-}=\frac{R_x-1}{R_x+1}.
  \]
  \begin{block}{Vocabulary}
    \small truth = generator level; reconstructed = from detector hits;
    folded = model passed through detector + reconstruction.
  \end{block}
\end{frame}
