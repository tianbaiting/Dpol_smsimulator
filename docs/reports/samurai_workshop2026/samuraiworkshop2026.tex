\section{Physics Case}

\begin{frame}{Motivation: Isovector Dynamics at Low Density}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.58\textwidth}
      \centering
      \includegraphics[width=0.84\linewidth]{Esym3.png}
      \vspace{0.05cm}

      {\scriptsize Prog. Part. Nucl. Phys. 134 (2024) 104080}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item The symmetry energy is still poorly constrained away from saturation density.
        \item Peripheral breakup of a polarized deuteron samples a dilute, asymmetric nuclear field.
        \item The experimental signature should be differential between proton and neutron momenta.
      \end{itemize}
      \vspace{0.25cm}
      \begin{block}{Goal}
        Use a final-state p/n momentum asymmetry as a clean handle on the isovector potential.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Isovector Reorientation Effect}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \centering
      \includegraphics[width=\linewidth]{isovector_reorientaton.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \small
      \begin{align*}
        E_{\mathrm{sym}}(\rho)
          &= \frac{C_{s,k}}{2}\left(\frac{\rho}{\rho_0}\right)^{2/3}
           + \frac{C_{s,p}}{2}\left(\frac{\rho}{\rho_0}\right)^{\gammastiff}, \\
        F_v &\propto \nabla U_{\mathrm{sym},1}
             \propto \frac{dE_{\mathrm{sym}}}{d\rho}.
      \end{align*}
      \begin{itemize}
        \item Neutron and proton feel opposite isovector forces near a neutron-rich target.
        \item For a polarized deuteron, this force pair acts as a torque on the p-n axis.
        \item The final p/n momentum difference carries the reorientation information.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Observable in the Reaction-Plane Frame}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=0.70\linewidth]{pol.pdf}

      \vspace{0.03cm}
      \includegraphics[width=0.70\linewidth]{unpol.pdf}
    \end{column}
    \begin{column}{0.52\textwidth}
      \scriptsize
      \begin{align*}
        \phi^{\mathrm{event}}
          &= \mathrm{atan2}(p_{y,\mathrm{p}}+p_{y,\mathrm{n}},
                            p_{x,\mathrm{p}}+p_{x,\mathrm{n}}), \\
        \dpx &= p_{x,\mathrm{p}}^{\mathrm{rot}} - p_{x,\mathrm{n}}^{\mathrm{rot}}, \\
        \Rx &= \frac{N(\dpx > 0)}{N(\dpx < 0)}.
      \end{align*}
      \begin{itemize}
        \setlength\itemsep{0.03cm}
        \item Rotate every event so $\vec p_T^{\,\mathrm{sum}}$ points along $+\hat{x}$.
        \item $\dpx$ isolates the p-n breakup asymmetry.
        \item $\Rx(\gammastiff)$ is the compact reconstruction benchmark.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Reconstruction Chain}

\begin{frame}{Why Reconstruction Is the Critical Step}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \centering
      \includegraphics[width=0.98\linewidth]{onlysamuraisetup.pdf}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item IVR is defined at the breakup target, but the experiment measures detector hits downstream.
        \item Proton momentum must be reconstructed from PDC1/PDC2 through the SAMURAI field.
        \item Neutron momentum must be reconstructed from NEBULA position and time of flight.
        \item The key feasibility test is whether $\Rx(\gammastiff)$ survives this detector and reconstruction chain.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{What Limits the Full Reconstruction?}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.5\textwidth}
      \begin{block}{Observed bottleneck order}
        \begin{center}
          \small
          NEBULA acceptance\\[-0.05cm]
          $\gg$ proton smearing $>$ neutron smearing
        \end{center}
      \end{block}
      \begin{itemize}
        \item PDC proton reconstruction is not the dominant limitation for $\Rx$.
        \item NEBULA loses part of the neutron phase space in the relevant $p_{x,n}$ window.
        \item A simple $1/\epsilon$ correction cannot recover events outside acceptance.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \includegraphics[width=0.92\linewidth]{reco_vs_truth_2d_ypol.png}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Interpretation for the Experiment}
  \begin{enumerate}
    \item The IVR-sensitive $\Rx(\gammastiff)$ trend survives detector simulation and proton target-momentum reconstruction.
    \item Sn124 is the clearest target in the present y-pol and z-pol samples.
    \item The proton arm is ready for physics-level analysis comparisons between RK and NN.
    \item The next systematic effort should focus on NEBULA visible-window strategy, reaction-plane reconstruction from reco momenta, and acceptance-limited $\Rx$ definitions.
  \end{enumerate}
\end{frame}

\begin{frame}{Summary}
  \begin{itemize}
    \item Isovector reorientation converts the neutron-proton force difference into an experimentally accessible breakup momentum asymmetry.
    \item The compact observable is $\Rx = N(\dpx>0)/N(\dpx<0)$ in the reaction-plane frame.
    \item NN proton reconstruction gives $\sim 6$ MeV/$c$ core resolution in $p_x$ and $p_z$ and preserves $\Rx(\gammastiff)$.
    \item y-pol Sn124 and z-pol Sn124 both retain strong monotonic $\gammastiff$ dependence after reconstruction.
    \item The dominant remaining issue is NEBULA acceptance/systematics, not PDC proton momentum reconstruction.
  \end{itemize}
\end{frame}

\section*{Backup}

\begin{frame}{Backup: Potential Link to Symmetry Energy}
  \small
  \begin{align*}
    E_{\mathrm{sym}}(\rho)
      &= \frac{1}{6}
        \frac{\partial [t(k)+U_0(\rho,k)]}{\partial k}\bigg|_{k_F} k_F
       + \frac{1}{2}U_{\mathrm{sym},1}(\rho,k_F), \\
      &= \frac{k_F^2}{6m_0^*(\rho,k_F)}
       + \frac{1}{2}U_{\mathrm{sym},1}(\rho,k_F).
  \end{align*}
  \begin{itemize}
    \item The potential part of $E_{\mathrm{sym}}$ is directly related to the isovector single-particle potential.
    \item The IVR torque is driven by the gradient of this isovector potential.
  \end{itemize}
  \[
    F_v \propto \nabla U_{\mathrm{sym},1}
        \propto \frac{dE_{\mathrm{sym}}}{d\rho}.
  \]
\end{frame}

\begin{frame}{Backup: NEBULA $\beta$ Reconstruction}
  \small
  \begin{align*}
    L &= |\vec r_{\mathrm{cluster}}-\vec r_{\mathrm{target}}|, \\
    \beta &= \frac{L/t}{c}, \\
    p_n &= m_n \frac{\beta}{\sqrt{1-\beta^2}}.
  \end{align*}
  \begin{itemize}
    \item Cluster position and time are energy-weighted from NEBULA hits.
    \item A target-frame rotation is applied after lab-frame reconstruction.
    \item Current simulation settings use 5 mm position smearing and 0.5 ns timing smearing.
    \item The $p_z$ bias is consistent with the sensitivity of $p=m\gamma\beta$ to time-of-flight errors and missing upstream corrections.
  \end{itemize}
\end{frame}
