\begin{frame}{What Are We Trying to Establish?}
  \small
  \begin{block}{Physics question}
    Can the isovector reorientation (IVR) signal from a y-polarized deuteron
    remain observable after the realistic SAMURAI detector and
    momentum-reconstruction chain?
  \end{block}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \begin{block}{Why two isotopes?}
        $^{112}$Sn provides an isotope reference.\\
        $^{124}$Sn provides the stronger neutron-rich response.
      \end{block}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{block}{Current answer}
        For $^{124}$Sn, the $\gpar$-dependent ordering survives reconstructed
        proton, neutron, and event-plane determination in a controlled neutron
        fiducial region.
      \end{block}
    \end{column}
  \end{columns}
  \begin{block}{Scope}
    The observable is fully reconstructed, while the present event-quality
    selection is still truth-assisted.
  \end{block}
  \centering
  \footnotesize
  This is a detector-level closure study, not yet a complete experimental
  uncertainty analysis.
\end{frame}

\begin{frame}{Physical Picture: Isovector Reorientation}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.42\textwidth}
      \centering
      \includegraphics[width=0.96\linewidth]{isovector_reorientaton.png}
    \end{column}
    \begin{column}{0.58\textwidth}
      \small
      \begin{itemize}
        \item A polarized deuteron approaches a neutron-rich target.
        \item Proton and neutron experience different isovector mean-field
          forces.
        \item The breakup fragments acquire a polarization-dependent relative
          momentum bias.
      \end{itemize}
      \[
        E_{\mathrm{sym}}(\rho)
        =
        E_{\mathrm{sym}}^{\mathrm{kin}}(\rho)
        + C\left(\frac{\rho}{\rho_0}\right)^{\gpar}
      \]
      \begin{block}{Model parameter}
        \small
        $\gpar$ parametrizes the density dependence used in the ImQMD
        calculation.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why Compare Two Tin Isotopes?}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \centering
      \begin{tabular}{lrrl}
        \toprule
        target & $Z$ & $N$ & role \\
        \midrule
        $^{112}$Sn & 50 & 62 & isotope reference \\
        $^{124}$Sn & 50 & 74 & primary sensitivity \\
        \bottomrule
      \end{tabular}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \item Same proton number controls much of the target charge and
          kinematic environment.
        \item Different neutron excess changes the isovector mean-field
          response.
        \item The same detector/reconstruction chain is required for a
          controlled isotope comparison.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.2cm}
  \begin{block}{Takeaway}
    A common detector effect should influence both isotopes, while an isovector
    response should change with target neutron excess.  $^{124}$Sn is the
    primary sensitivity target; $^{112}$Sn is the isotope reference.
  \end{block}
\end{frame}

\begin{frame}{Why Start with y Polarization?}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item y-pol and z-pol are both physically valuable IVR channels.
        \item y-pol currently has the most mature combined simulation,
          detector folding, momentum reconstruction, and analysis baseline.
        \item y-pol also has an established left-right-up-down (LRUD)
          beam-monitoring strategy for vertical tensor polarization $p_{yy}$.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{First-stage baseline}
        y-pol provides a realistic first-stage path to testing the IVR concept at
        SAMURAI.
      \end{block}
      \begin{block}{z-pol}
        z-pol remains a second-stage extension after successful commissioning.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Reconstructed IVR Observable}
  \small
  \[
    \phi_{\mathrm{event}}^{\mathrm{reco}}
    =
    \operatorname{atan2}
    \left(
      p_{y,p}^{\mathrm{reco}}+p_{y,n}^{\mathrm{reco}},
      p_{x,p}^{\mathrm{reco}}+p_{x,n}^{\mathrm{reco}}
    \right)
  \]
  \[
    \dpxreco =
    p_{x,p,\mathrm{reco}}^{\mathrm{rot}}
    -
    p_{x,n,\mathrm{reco}}^{\mathrm{rot}}
  \]
  \[
    \Rxreco
    =
    \frac{N(\dpxreco>0)}
         {N(\dpxreco<0)}
  \]
  \begin{block}{Experimental comparison}
    \[
      R_{x,\mathrm{data}}^{\mathrm{reco,fid}}
      \quad\text{vs.}\quad
      \Rxfold(\gpar)
    \]
  \end{block}
  \centering
  \footnotesize
  This is a reconstructed fiducial observable, not a global truth-level ratio.
\end{frame}

\begin{frame}{Statistical Scale versus Experimental Systematics}
  \small
  Closest adjacent interval in the current $^{124}$Sn y-pol folded result:
  \[
    \gpar=0.7\to0.8,\qquad
    N_{\mathrm{stat},3\sigma}\simeq 2.8\times10^3 .
  \]
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Planning scale}
        Current planning anchor gives about $2.75\times10^5$ usable y-pol
        events for $^{124}$Sn.
      \end{block}
    \end{column}
    \begin{column}{0.52\textwidth}
      \begin{block}{Not included}
        Detector, polarization, background, and model uncertainties are not
        included in this statistical-only estimate.
      \end{block}
    \end{column}
  \end{columns}
  \begin{block}{Systematic priorities}
    neutron efficiency; threshold and time-of-flight $T_0$; cross-talk and fake
    neutrons; backgrounds; $p_{yy}$ polarization; detector alignment; model
    dependence; reco-only selection migration.
  \end{block}
\end{frame}

\begin{frame}{Current Status and Next Closure Step}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \begin{block}{Established}
        \begin{itemize}
          \item y-pol IVR observable is reconstructable.
          \item $^{124}$Sn $\gpar$ ordering survives detector folding.
          \item px60 provides a controlled fiducial region.
          \item folded model comparison is the right strategy.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{block}{Still required}
        \begin{itemize}
          \item reco-only event-quality cuts,
          \item reco-only Sn112/Sn124 isotope closure,
          \item Sn112 target-specific rate validation,
          \item selection efficiency, purity, and migration,
          \item polarization and detector-systematic pseudo-data tests.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \centering
  \footnotesize
  The next decisive analysis is not another truth-level calculation. It is a
  fully reco-only isotope closure test.
\end{frame}

\begin{frame}{Summary}
  \small
  \begin{enumerate}
    \item IVR produces a polarization-dependent proton-neutron relative-momentum
      bias near a neutron-rich target.
    \item $^{124}$Sn is the primary sensitivity target, while $^{112}$Sn
      provides a controlled isotope reference.
    \item The $\gpar$-dependent ordering survives realistic detector folding and
      full proton, neutron, and event-plane reconstruction in the px60 fiducial
      region for $^{124}$Sn.
    \item The experiment should compare reconstructed data with identically
      folded model predictions; the next step is reco-only isotope closure and
      detector-systematic validation.
  \end{enumerate}
  \begin{block}{Bottom line}
    The present study establishes detector-level observability, not yet a
    complete experimental uncertainty budget.  A y-pol measurement provides a
    realistic first-stage path to testing the IVR concept at SAMURAI.
  \end{block}
\end{frame}

\begin{frame}{Backup A: Observable and Bounded Asymmetry}
  \small
  \[
    \dpxreco =
    p_{x,p,\mathrm{reco}}^{\mathrm{rot}}
    -
    p_{x,n,\mathrm{reco}}^{\mathrm{rot}},
    \qquad
    \Rxreco =
    \frac{N(\dpxreco>0)}{N(\dpxreco<0)} .
  \]
  \[
    \Ax=\frac{N_+-N_-}{N_++N_-}
       =\frac{R_x-1}{R_x+1}.
  \]
  \begin{block}{Vocabulary}
    Truth, reconstructed, folded, and unfolded observables are different
    objects.  The main result uses a reconstructed fiducial observable.
  \end{block}
\end{frame}

\begin{frame}{Backup A: Folding Vocabulary}
  \small
  \[
    N_{\mathrm{reco},j}=\sum_iK_{ji}N_{\mathrm{true},i}+b_j .
  \]
  \begin{itemize}
    \item truth: generator-level kinematics after physics selection,
    \item reconstructed: variables built from detector hits,
    \item folded: model events passed through detector response and
      reconstruction,
    \item unfolded: estimator of truth after response inversion and
      regularization.
  \end{itemize}
  \begin{block}{Current limitation}
    The event-quality cut is still truth-assisted; the neutron px60 fiducial and
    the observable numerator/denominator are reconstructed.
  \end{block}
\end{frame}
