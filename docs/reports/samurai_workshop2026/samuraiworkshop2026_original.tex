\section{Physics Motivation}

\begin{frame}{Motivation: Symmetry Energy at Low Density}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.58\textwidth}
      \centering
      \includegraphics[width=0.84\linewidth]{Esym3.png}
      \vspace{0.05cm}

      {\scriptsize Prog. Part. Nucl. Phys. 134 (2024) 104080}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item The symmetry energy is still poorly constrained away from saturation density.
        \item Peripheral breakup of a polarized deuteron samples a dilute, asymmetric nuclear field.
        \item The experimental signature is a differential proton--neutron momentum distortion.
      \end{itemize}
      \vspace{0.2cm}
      \begin{block}{Goal}
        Use a final-state p/n momentum asymmetry as a clean handle on the isovector potential, parametrized by the stiffness $\gammastiff$.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Isovector Reorientation Effect}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.55\textwidth}
      \centering
      \includegraphics[width=\linewidth]{isovector_reorientaton.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \small
      \begin{align*}
        E_{\mathrm{sym}}(\rho)
          &= \frac{C_{s,k}}{2}\left(\frac{\rho}{\rho_0}\right)^{2/3}
            + \frac{C_{s,p}}{2}\left(\frac{\rho}{\rho_0}\right)^{\gammastiff}, \\
        F_v &\propto \nabla U_{\mathrm{sym},1}
             \propto \frac{dE_{\mathrm{sym}}}{d\rho}.
      \end{align*}
      \begin{itemize}
        \item Neutron and proton feel opposite isovector forces near a neutron-rich target.
        \item For a polarized deuteron, this force pair acts as a torque on the p--n axis.
        \item A stiffer symmetry energy ($\uparrow\gammastiff$) changes the magnitude of this torque.
        \item The final p/n momentum difference carries the reorientation information.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Observable}

\begin{frame}{Observable in the Reaction-Plane Frame}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=0.70\linewidth]{pol.pdf}

      \vspace{0.03cm}
      \includegraphics[width=0.70\linewidth]{unpol.pdf}
    \end{column}
    \begin{column}{0.52\textwidth}
      \scriptsize
      Define the event reaction plane from the outgoing p+n transverse momentum:
      \[
        \phi^{\mathrm{event}}
          = \mathrm{atan2}(p_{y,\mathrm{p}}+p_{y,\mathrm{n}},
                            p_{x,\mathrm{p}}+p_{x,\mathrm{n}}).
      \]
      Rotate every event so $\vec p_T^{\,\mathrm{sum}}$ points along $+\hat{x}$, then
      \[
        \dpx = p_{x,\mathrm{p}}^{\mathrm{rot}} - p_{x,\mathrm{n}}^{\mathrm{rot}},
        \qquad
        \Rx = \frac{N(\dpx > 0)}{N(\dpx < 0)}.
      \]
      \begin{itemize}
        \setlength\itemsep{0.03cm}
        \item $\dpx$ isolates the p-n breakup asymmetry.
        \item $\Rx(\gammastiff)$ is the compact observable; it is sensitive to the isovector torque.
        \item The experiment must form $\dpx$ from \emph{reconstructed} proton and neutron momenta.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Honest Caveat: Event Selection}
  \small
  \begin{block}{What is fully reconstructed}
    \small The $\Rx$ observable itself uses \emph{reconstructed} proton momentum (NN), \emph{reconstructed} neutron momentum (NEBULA + NEBULA Plus $\beta$ method), and the \emph{reconstructed} event reaction plane. No truth momentum enters the numerator or denominator of $\Rx$.
  \end{block}
  \begin{block}{What is still truth-defined}
    \small The event-quality selection (\texttt{tight}: collinearity and transverse-momentum balance) is currently defined on truth momenta. The visible window $|p_{x,n}^{\mathrm{reco}}|<60$ MeV/$c$ \emph{is} reconstructed.
  \end{block}
  \begin{block}{Implication}
    \small The present result is a \emph{detector-level closure test}: it demonstrates that the $\gammastiff$ dependence survives reconstruction. The final experimental analysis must migrate the event-quality cuts to reconstructed quantities.
  \end{block}
\end{frame}

\section{Target Comparison and NEBULA Plus Role}

\begin{frame}{Target Comparison}
  \small
  \begin{itemize}
    \item The NEBULA Plus production analysis presented here focuses on $^{124}\mathrm{Sn}$, which at truth level shows the strongest and most monotonic $\Rx(\gammastiff)$ dependence.
    \item $^{112}\mathrm{Sn}$ is expected to be weaker and less monotonic (fewer excess neutrons $\Rightarrow$ smaller isovector field).
    \item A full NEBULA Plus reconstruction pass for $^{112}\mathrm{Sn}$ with the same px60 visible-window definition is pending; its truth-level numbers should not be mixed with the $^{124}\mathrm{Sn}$ NEBULA Plus result.
  \end{itemize}
  \vspace{0.15cm}
  \begin{block}{Current statement}
    \small $^{124}\mathrm{Sn}$ is the primary sensitivity target. Its reconstructed $\Rx(\gammastiff)$ remains monotonic and well separated after the full detector chain including NEBULA Plus.
  \end{block}
\end{frame}

\begin{frame}{What NEBULA Plus Enables}
  \small
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \item NEBULA Plus adds neutron bars beyond the base NEBULA wall, increasing the angular coverage for forward neutrons.
        \item The neutron candidate pool is the \emph{union} of NEBULA and NEBULA Plus hits; the joint reconstruction provides the sample from which $\Rx$ is built.
        \item Without the combined acceptance there would not be enough reconstructable neutron events in the px60 visible window to form a stable $\Rx$.
      \end{itemize}
      \begin{block}{Defensible statement}
        \scriptsize The present result demonstrates observability of the IVR signal in the detector configuration \emph{including} NEBULA Plus.
      \end{block}
    \end{column}

\section{Summary}

\begin{frame}{Summary}
  \small
  \begin{enumerate}
    \item Truth-level isovector reorientation produces a strong $\Rx(\gammastiff)$ dependence for $^{124}\mathrm{Sn}$.
    \item Proton target-momentum reconstruction (NN, cross-checked against RK) preserves the relevant momentum asymmetry ($\sim$6 MeV/$c$ core resolution).
    \item NEBULA + NEBULA Plus provides a reconstructable neutron sample with a modelable ($\sim$5\%) sign-dependent efficiency imbalance in the px60 window.
    \item \textbf{After the full detector response and NEBULA Plus reconstruction, the detector-level $\Rx$ still shows a clear, monotonic $\gammastiff$ dependence} (y-pol $12.48\!\to\!4.99$, z-pol $6.38\!\to\!0.56$ for $\gammastiff\!=\!0.5\!\to\!0.8$).
    \item The IVR physics effect therefore remains experimentally observable in the full SAMURAI + NEBULA Plus configuration.
  \end{enumerate}
  \vspace{0.15cm}
  \begin{block}{Bottom line}
    \small The main result is not that reconstruction exactly reproduces the truth-level ratio. The main result is that the $\gammastiff$-dependent separation survives the full detector and reconstruction chain including NEBULA Plus.
  \end{block}
\end{frame}

\begin{frame}{Backup: Observable Definition and Visible Window}
  \small
  The experiment does not observe the truth distribution directly:
  \[
    N_{\mathrm{reco},j} = \sum_i K_{ji}\, N_{\mathrm{true},i} + b_j,
    \qquad R_{\mathrm{reco}} \ne R_{\mathrm{truth}}.
  \]
  Here $K_{ji}$ includes NEBULA/NEBULA Plus acceptance, resolution, reconstruction efficiency, PDC proton reconstruction, reaction-plane migration, and cuts.
  \begin{itemize}
    \item The robust strategy is to fold every theory point through the same Geant4 + reconstruction chain before comparing with data.
    \item The px60 visible window keeps the region where both $p_{x,n}$ signs have $>50\%$ neutron efficiency.
    \item Statistical error: $\sigma(\Rx) = \Rx\sqrt{1/N_+ + 1/N_-}$.
  \end{itemize}
\end{frame}

\begin{frame}{Backup: Potential Link to Symmetry Energy}
  \small
  \begin{align*}
    E_{\mathrm{sym}}(\rho)
      &= \frac{1}{6}
        \frac{\partial [t(k)+U_0(\rho,k)]}{\partial k}\bigg|_{k_F} k_F
       + \frac{1}{2}U_{\mathrm{sym},1}(\rho,k_F), \\
      &= \frac{k_F^2}{6m_0^*(\rho,k_F)}
       + \frac{1}{2}U_{\mathrm{sym},1}(\rho,k_F).
  \end{align*}
  \begin{itemize}
    \item The potential part of $E_{\mathrm{sym}}$ is directly related to the isovector single-particle potential.
    \item The IVR torque is driven by the gradient of this isovector potential:
      $F_v \propto \nabla U_{\mathrm{sym},1} \propto dE_{\mathrm{sym}}/d\rho$.
  \end{itemize}
\end{frame}
