\begin{frame}{Study Goal and Configuration}
\textbf{Goal}: Evaluate reconstruction quality with synthetic beam input and focus on
\[
\Delta P_x = P_x^{\mathrm{reco}} - P_x^{\mathrm{true}} .
\]

\vspace{0.3cm}
\textbf{Configuration}
\begin{itemize}
\item Magnetic setting from macro: \texttt{run\_protons\_B115T\_3.0deg.mac}
\item Geometry/field macro: \texttt{geometry\_B115T.mac}
\item Target from macro: \((-12.488, 0.009, -1069.458)\,\mathrm{mm}\), angle \(3.0^\circ\)
\item Dipole field map: \texttt{180703-1,15T-3000.table}, rotation \(30^\circ\)
\end{itemize}
\end{frame}

\begin{frame}{Synthetic Input Dataset}
\textbf{Generator setup}
\begin{itemize}
\item Proton momentum: \((x_p, 0, 627)\,\mathrm{MeV}/c\), \(x_p \sim \mathcal{U}[-150,150]\)
\item Neutron momentum: \((x_n, 0, 627)\,\mathrm{MeV}/c\), \(x_n \sim \mathcal{U}[-150,150]\)
\item Generated events: 20,000
\item Reconstruction analysis sample: 3,000 events
\end{itemize}

\vspace{0.2cm}
\textbf{Event statistics in analyzed sample}
\begin{itemize}
\item Events with reconstructed PDC track: 2597 / 3000 (\(86.57\%\))
\item Events with neutron TOF reconstruction: 994 / 3000 (\(33.13\%\))
\item Proton fit sample cap: 600 events (Minuit/Three-point branches)
\item Three-point fit convergence: \(362/600 = 60.33\%\)
\end{itemize}
\end{frame}

\begin{frame}{PDC Reconstruction Process (Charged Proton)}
\textbf{Input}
\begin{itemize}
\item \texttt{FragSimData} PDC hits (\(U/V\) layers for PDC1/PDC2).
\item Geometry from \texttt{geometry\_B115T.mac} (PDC positions/angles + target transform).
\end{itemize}

\textbf{Reconstruction chain}
\begin{itemize}
\item Convert \(U/V\) hits to CoM points in each chamber, with configured smearing.
\item Build one straight \texttt{RecoTrack} with two points:
\[
\vec{r}_1=\text{PDC1 reco point},\quad \vec{r}_2=\text{PDC2 reco point}.
\]
\item Target back-tracing in field map (\(B=1.15\)T):
\begin{itemize}
\item Method A: \textbf{TMinuit} RK-based target momentum fit.
\item Method B: \textbf{Three-point (free-parameter)} fit with 6D optimization
\((x_0,y_0,z_0,p_{x0},p_{y0},p_{z0})\), constrained by target/PDC1/PDC2 distances.
\end{itemize}
\item Validation observable:
\[
\Delta P_x = P_x^{\mathrm{reco}} - P_x^{\mathrm{true}}.
\]
\end{itemize}
\end{frame}

\section{Neural Network Target-Momentum Reconstruction}

\begin{frame}{NN Objective and Scope}
\textbf{Objective}
\begin{itemize}
\item Reconstruct target momentum directly:
\[
(\hat p_x,\hat p_y,\hat p_z)=f(\text{PDC two-point track}).
\]
\item This branch predicts momentum components, not \(B\rho\), and does not use matrix inversion.
\end{itemize}

\textbf{Scope in this revision}
\begin{itemize}
\item Offline ML pipeline only (dataset/training/inference/evaluation scripts).
\item Not yet integrated into the online C++ auto-chain backend.
\end{itemize}
\end{frame}

\begin{frame}{Dataset Construction}
\textbf{Sources}
\begin{itemize}
\item Simulation tree branches: \texttt{FragSimData} and \texttt{beam}.
\item Geometry/coordinate transform from selected macro.
\end{itemize}

\textbf{Features and labels}
\begin{itemize}
\item Input \(X\in\mathbb{R}^6\): \(\texttt{pdc1\_x,y,z}\), \(\texttt{pdc2\_x,y,z}\).
\item Target \(Y\in\mathbb{R}^3\): \((p_x^{true},p_y^{true},p_z^{true})\) of proton truth momentum in target frame.
\end{itemize}

\textbf{Filtering and outputs}
\begin{itemize}
\item Keep events with proton truth, valid reconstructed PDC track, and finite feature/label values.
\item Output files: \texttt{dataset\_train.root} and \texttt{dataset\_train.csv}.
\item Builder script: \texttt{scripts/reconstruction/nn\_target\_momentum/build\_dataset.C}.
\end{itemize}
\end{frame}

\begin{frame}{Train/Validation/Test Split and Model}
\textbf{Split policy}
\begin{itemize}
\item Formal campaign uses three independent simulation runs (train/val/test).
\item Seeds: train \texttt{20260301}, val \texttt{20260302}, test \texttt{20260303}.
\item \texttt{train\_mlp.py} consumes external split files (\texttt{--train-csv --val-csv --test-csv}).
\end{itemize}

\textbf{Model and training}
\begin{itemize}
\item MLP regressor: \(6 \rightarrow 128 \rightarrow 128 \rightarrow 64 \rightarrow 3\) (ReLU).
\item Input z-score normalization from training split statistics.
\item Optimizer/loss: Adam + MSE with early stopping.
\item Training script: \texttt{scripts/reconstruction/nn\_target\_momentum/train\_mlp.py}.
\end{itemize}
\end{frame}

\begin{frame}{Evaluation and Current Limitations}
\textbf{Where to read official hold-out metrics}
\begin{itemize}
\item Hold-out metrics are reported from dedicated val/test datasets.
\item Files: \texttt{predictions/metrics\_val.json} and \texttt{predictions/metrics\_test.json}.
\item Training metadata also stores \texttt{val\_metrics} and \texttt{test\_metrics} in \texttt{model\_meta.json}.
\end{itemize}

\textbf{Generalization note}
\begin{itemize}
\item Inputs currently include only PDC two-point coordinates.
\item Beam/field configuration is learned implicitly from the training data distribution.
\item Validity remains configuration-dependent; retraining is required when geometry/field changes.
\item Effective dataset size is limited by PDC track availability after reconstruction cuts.
\end{itemize}
\end{frame}

\begin{frame}{NEBULA Reconstruction Process (Neutron TOF)}
\textbf{Input}
\begin{itemize}
\item \texttt{NEBULAPla} hit array (position, time, deposited energy).
\item Target point from loaded geometry macro.
\end{itemize}

\textbf{Reconstruction chain}
\begin{itemize}
\item Time-window hit clustering (\(10\) ns) and energy threshold cut (\(1\) MeV).
\item Pick best neutron candidate (highest reconstructed energy cluster).
\item Flight vector:
\[
\vec{L}=\vec{r}_{\mathrm{hit}}-\vec{r}_{\mathrm{target}},\quad L=|\vec{L}|.
\]
\item TOF momentum:
\[
\beta=\frac{L/t}{c},\quad p=\gamma m_n \beta c,\quad \vec{p}=p\cdot\frac{\vec{L}}{L}.
\]
\item Validation observable:
\[
\Delta P_x = P_x^{\mathrm{reco}} - P_x^{\mathrm{true}}.
\]
\end{itemize}
\end{frame}

\begin{frame}{Conclusions and Next Actions}
\textbf{Conclusions}
\begin{itemize}
\item Uniform-incidence proton input is not reconstructed uniformly at high \(+P_x\) due to PDC acceptance edge.
\item Neutron TOF branch remains stable: Mean(\(\Delta P_x\)) \(=+1.758\) MeV/c, RMS \(=17.152\) MeV/c (994 events).
\item Proton target back-tracing is method-dependent: Minuit has large negative bias; free three-point reaches Mean(\(\Delta P_x\)) \(=+6.180\) MeV/c, RMS \(=39.540\) MeV/c with \(60.33\%\) convergence.
\end{itemize}

\textbf{Immediate next actions}
\begin{itemize}
\item Keep three-point priors and tune convergence from \(60\%\) toward \(>90\%\).
\item Fix two-point Minuit sign/bias issue (field/charge/direction convention check).
\item Audit PDC acceptance edge (\(+P_x\)) with chamber-level hit occupancy.
\end{itemize}
\end{frame}
