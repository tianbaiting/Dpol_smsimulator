\section{cut on unphysical breakup}


\begin{frame}{unphysical breakup}
    \begin{itemize}
        \item \textbf{Issue}: QMD simulations show unphysical breakup events where the relative momentum between proton and neutron exceeds 200 MeV/c.
        \item \textbf{Solution}: Apply a cut to remove these events:
        \begin{align*}
            p^p_{i'} - p^n_{i'} \in (-150, 150) \mathrm{MeV/c}
        \end{align*}
        where \(i'\) is the polarization direction (z for z-polarization, y for y-polarization).
        \item \textbf{Reaction-plane definition}: rotate each event around beam \(z\) axis by \(-\phi\), with \(\phi=\tan^{-1}\!\left((p^p_y+p^n_y)/(p^p_x+p^n_x)\right)\), so that \(P_x^{\mathrm{tot}}\) is aligned in the reaction plane.
        \item \textbf{Observable}:
        \[
        R=\frac{N(P_x^p>P_x^n)}{N(P_x^p<P_x^n)}
        \]
    \end{itemize}
\end{frame}


\section{obersevation R }


\begin{frame}{Our Probe: The Isovector Reorientation (IVR) Effect}
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
        \centering
    \includegraphics[width=\columnwidth]{./img/isovector_reorientaton.png}
    \end{column}
    \begin{column}{0.4\textwidth}
    \tiny{
    \begin{align*}
        E_{\text{sym}}(\rho) = \frac{C_{s,k}}{2} \left(\frac{\rho}{\rho_0}\right)^{2/3} + \\ \frac{C_{s,p}}{2} \left(\frac{\rho}{\rho_0}\right)^{\gamma}
    \end{align*}
    }


    $$
    \small
    \gamma \uparrow \Rightarrow \frac{dE_{\text{sym}}}{d\rho} \downarrow \Rightarrow F_v \downarrow
    $$
    \end{column}
\end{columns}

    \begin{itemize}
        \item[\textcolor{blue}{$\blacklozenge$}] In peripheral scattering of a polarized deuteron, the \textbf{isovector nuclear force} acts as a torque.
        \begin{itemize}
            \item Attractive for protons
            \item Repulsive for neutrons
        \end{itemize}
        \item[\textcolor{blue}{$\blacklozenge$}] This causes an "extra rotation" of the deuteron before it breaks up.
        \item[\textcolor{blue}{$\blacklozenge$}] This makes it a \textbf{clean probe}: sensitive to the isovector potential (\(E_{\text{sym}}\)) but insensitive to the isoscalar potential.
    \end{itemize}


    %  这里加入 Px+Px意义， Px-Px.
\end{frame}

\begin{frame}{Recommended single-particle $p_x$ windows}

\begin{tabular}{c c c c}
\toprule
Dataset & Particle & Suggested \(p_x\) window (MeV/c) & \(\Delta R_\gamma\) (spread) \\
\midrule
Y-pol & proton & [25, 50] & 31.97 \\
Y-pol & neutron & [0, 25] & 5.40 \\
Z-pol & proton & [25, 50] & 6.53 \\
Z-pol & neutron & [0, 25] & 1.61 \\
\bottomrule
\end{tabular}
\vspace{0.2cm}

\begin{itemize}
    \item Sensitivity ranking is done by \(\max_\gamma(R)-\min_\gamma(R)\) with high-statistics requirement (\(N_{\mathrm{bin}}\ge 2000\)).
    \item For current Pb208 sample, proton \(p_x\in[25,50]\) MeV/c is the strongest region in both y-pol and z-pol.
    \item Neutron sensitivity is weaker and concentrated near \(p_x\in[0,25]\) MeV/c.
\end{itemize}
\end{frame}
